Usage examples

Asynchronous Module Definition

<script src="https://cdnjs.cloudflare.com/ajax/libs/require.js/2.3.6/require.min.js"></script>
<script>
requirejs.config({
  "paths": {
    "example": "../dist/example.amd"
  }
});
requirejs(["example"], ({hello}) => {
  hello("Famichiki");
});
</script>

CommonJS (Node.js)

// CommonJS does not work on browser.
const {hello} = require("../dist/example.cjs.js");

hello("Famichiki");

ES Modules

<script type="module">
import {hello} from "../dist/example.esm.js";

hello("Famichiki");
</script>

IIFE(Immediately Invoked Function Expression)

<script src="../dist/example.iife.js"></script>
<script>
Example.hello("Famichiki");
</script>

Universal Module Definition

<script src="../dist/example.umd.js"></script>
<script>
Example.hello("Famichiki");
</script>
